import re

# ── Fix 1: Chat.tsx ──────────────────────────────────────────────────────────
chat = "artifacts/voice-companion/src/pages/Chat.tsx"
with open(chat) as f:
    c = f.read()

# 1a. Lower sentence threshold 20→8 chars, handle closing quotes after terminal
c = c.replace(
    '/^.{20,}?[.!?](?=\\s)/s',
    '/^.{8,}?[.!?]["\\\']?(?=\\s|$)/s'
)

# 1b. Stop stripping ALL parentheticals — only drop known stage directions
OLD_PAREN = '.replace(/\\([^)]*\\)/g, "")'
NEW_PAREN = '.replace(/\\((?:laughs?|chuckles?|sighs?|gasps?|smiles?|grins?|pauses?|whispers?|softly|quietly|nervously|warmly|teasingly|playfully|gently|hesitates?|nods?)[^)]*\\)/gi, "")'
assert OLD_PAREN in c, "parenthetical regex not found — check Chat.tsx"
c = c.replace(OLD_PAREN, NEW_PAREN)   # replaces both occurrences (leg-1 inline + cleanTTS)

# 1c. Re-detect sentence boundary in authoritative full_text before slicing
SLICE_ANCHOR = 'remainingSpoken = cleanTTS(fullReply.slice(firstSentenceEndIdx))'
assert SLICE_ANCHOR in c, "remainingSpoken slice anchor not found"
REDETECT = (
    '// Re-sync index to authoritative full_text so accumulated-token drift '
    'can\'t truncate leg 2\n'
    '      if (event.full_text) {\n'
    '        const _sm = event.full_text.match(/^.{8,}?[.!?]["\\\']?(?=\\s|$)/s);\n'
    '        if (_sm) firstSentenceEndIdx = _sm[0].length;\n'
    '      }\n'
    '      '
)
c = c.replace(SLICE_ANCHOR, REDETECT + SLICE_ANCHOR)

with open(chat, 'w') as f:
    f.write(c)
print("✓ Chat.tsx — threshold lowered, parenthetical strip narrowed, index re-synced")
